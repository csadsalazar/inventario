-- Crear base de datos
IF EXISTS (SELECT * FROM sys.databases WHERE name = N'Gestion_inventario')
    DROP DATABASE Gestion_inventario
GO

CREATE DATABASE Gestion_inventario
GO

USE Gestion_inventario
GO

-- Tabla de Usuarios
CREATE TABLE MA_Usuarios (
    PK_idUsuario INT PRIMARY KEY IDENTITY,
    nombre NVARCHAR(50),
    usuario NVARCHAR(15),
    cedula INT,
    contrasena NVARCHAR(20),
    dependencia INT NOT NULL,
    cargo NVARCHAR(20) NOT NULL,
    contrato NVARCHAR(20) NOT NULL,
    sede NVARCHAR(30) NOT NULL
)

-- Tabla de Bienes
CREATE TABLE MA_Bienes (
    idBien INT PRIMARY KEY IDENTITY,
    PK_Codigo INT,
    nombre NVARCHAR(50),
    placa NVARCHAR(6),
    descripcion NVARCHAR(150),
    valor INT,
    FK_Usuario INT,
    FK_Dependencia INT,
    estado NVARCHAR(12),
    imagen1 NVARCHAR(50),
    imagen2 NVARCHAR(50),
    FOREIGN KEY (FK_Usuario) REFERENCES MA_Usuarios(PK_idUsuario) ON DELETE CASCADE 
)

-- Tabla de Bienes por Usuario
CREATE TABLE PA_BienesPorUsuario (
    PK_idBienPorUsuario INT PRIMARY KEY IDENTITY,   
    FK_Usuario INT,
    FK_Bien INT,
    asunto NVARCHAR(25),
    informacion NVARCHAR(200),
    fechaObservacion DATE NOT NULL,
    FOREIGN KEY (FK_Usuario) REFERENCES MA_Usuarios(PK_idUsuario), 
    FOREIGN KEY (FK_Bien) REFERENCES MA_Bienes(idBien) ON DELETE CASCADE
)

-- Tabla de Usuarios Independiente
CREATE TABLE MA_Administradores (
    PK_idAdministrador INT PRIMARY KEY IDENTITY,
    usuario NVARCHAR(15),
    estado NVARCHAR(8)
)

CREATE TABLE MA_Dependencias (
    PK_idDependencia INT PRIMARY KEY IDENTITY,
    nombreDependencia NVARCHAR(130) UNIQUE
)

-- Procedimientos almacenados para Bienes
CREATE PROCEDURE AgregarBien (
    @codigo INT,
    @nombre NVARCHAR(50),
    @placa NVARCHAR(6),
    @descripcion NVARCHAR(150),
    @valor INT,
    @FK_Usuario INT,
    @FK_Dependencia INT,
    @estado NVARCHAR(12),
    @imagen1 NVARCHAR(50),
    @imagen2 NVARCHAR(50)
)
AS
BEGIN
    INSERT INTO MA_Bienes (PK_Codigo, nombre, placa, descripcion, valor, FK_Usuario, FK_Dependencia, estado, imagen1, imagen2)
    VALUES (@codigo, @nombre, @placa, @descripcion, @valor, @FK_Usuario, @FK_Dependencia, @estado, @imagen1, @imagen2)
END

CREATE PROCEDURE ListarBienes
AS
BEGIN
    SELECT b.idBien,
           b.PK_Codigo,
           b.placa,
           b.nombre,
           u.usuario,
           u.dependencia,
           b.valor 
    FROM MA_Bienes b 
    INNER JOIN MA_Usuarios u ON b.FK_Usuario = u.PK_idUsuario
END

CREATE PROCEDURE AgregarObservacion (
    @usuario INT,
    @bien INT,
    @asunto NVARCHAR(50),
    @informacion NVARCHAR(250), 
    @fechaObservacion DATE
)
AS
BEGIN
    INSERT INTO PA_BienesPorUsuario (FK_Usuario, FK_Bien, asunto, informacion, fechaObservacion)
    VALUES (@usuario, @bien, @asunto, @informacion, @fechaObservacion)
END

CREATE PROCEDURE ListarObservaciones
AS
BEGIN
    SELECT bu.FK_Usuario,
           u.PK_idUsuario,
           u.usuario,
           u.dependencia,
           bu.asunto,
           bu.informacion 
    FROM PA_BienesPorUsuario bu 
    INNER JOIN MA_Usuarios u ON bu.FK_Usuario = u.PK_idUsuario
END
